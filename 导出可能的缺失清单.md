### **立即补充（Phase 3.1）**：

1. **rt_subscr.cpp 新增快速路径**：
   ```cpp
   PYLANG_EXPORT_SUBSCR("list_getitem_fast", "obj", "obj,i32")
   py::PyObject *rt_list_getitem_fast(py::PyObject *list, int32_t index)
   {
       return rt_unwrap(static_cast<py::PyList *>(list)->__getitem__(static_cast<int64_t>(index)));
   }
   
   PYLANG_EXPORT_SUBSCR("dict_getitem_str", "obj", "obj,str")
   py::PyObject *rt_dict_getitem_str(py::PyObject *dict, const char *key)
   {
       auto *key_str = rt_unwrap(py::PyString::create(std::string(key)));
       return rt_unwrap(dict->getitem(key_str));
   }
   ```

2. **rt_convert.cpp 新增类型转换**：
   ```cpp
   PYLANG_EXPORT_CONVERT("len", "i64", "obj")
   int64_t rt_len(py::PyObject *obj)
   {
       auto seq = obj->as_sequence();
       if (seq.is_ok()) {
           return rt_unwrap(seq.unwrap().len());
       }
       auto map = obj->as_mapping();
       if (map.is_ok()) {
           return rt_unwrap(map.unwrap().len());
       }
       rt_raise(py::type_error("object of type '{}' has no len()", obj->type()->name()));
   }
   
   PYLANG_EXPORT_CONVERT("isinstance", "bool", "obj,obj")
   bool rt_isinstance(py::PyObject *obj, py::PyObject *type)
   {
       auto *type_obj = py::as<py::PyType>(type);
       if (!type_obj) {
           rt_raise(py::type_error("isinstance() arg 2 must be a type"));
       }
       return obj->type()->issubclass(type_obj);
   }
   ```

3. **`rt_builtin.cpp` 新增（最小集）**：
   ```cpp
   PYLANG_EXPORT_BUILTIN("print", "void", "obj")
   void rt_print(py::PyObject *obj)
   {
       auto str = obj->str();
       if (str.is_err()) { rt_raise(str.unwrap_err()); }
       std::cout << str.unwrap()->value() << std::endl;
   }
   
   PYLANG_EXPORT_BUILTIN("repr", "obj", "obj")
   py::PyObject *rt_repr(py::PyObject *obj)
   {
       return rt_unwrap(obj->repr());
   }
   ```
